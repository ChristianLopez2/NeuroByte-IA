RAG Licitaciones – Backend FastAPI

Este módulo implementa un backend para analizar licitaciones usando un enfoque RAG (Retrieval Augmented Generation). Permite:

  • Indexar documentos base (normativa/modelos) y documentos subidos (ofertas) en ChromaDB.
  • Extraer texto de PDFs con PyMuPDF y OCR (Tesseract / EasyOCR opcional).
  • Detectar metadatos de proceso (familia/procedimiento/doc_role) a través de patrones YAML.
  • Consultar el SRI para validar RUCs de oferentes.
  • Construir prompts robustos con “candidatos confiables” (anti-alucinación) para el LLM.
  • Responder comparativos y JSON de decisión normalizado por oferta.
  • Persistir resultados por “tipo de proceso” en /static/tipos/*.json.

Estructura principal

  • Settings: rutas, modelos, opciones de chunking, top‑K, etc.
  • OCR/PDF utils: extracción de texto eficiente con “head” para clasificación rápida.
  • KBConfig: lectura de knowledge/kb_config.yaml con facets (familia, procedimiento, rol).
  • Vector stores: Chroma para BASE y UPLOADS.
  • SRI/RUC: utilidades para normalizar y validar estado del contribuyente.
  • Extractores: regex y heurísticas anti-alucinación (IDs, montos, plazos, cláusulas).
  • Prompt unificado: especifica formato de salida y reglas de decisión.
  • Endpoints: /api/upload-pdf, /api/chat, /api/tipos-licitacion, etc.
  • Persistencia: guarda respuesta2 por “destino” en /static/tipos/*.json con IDs únicos.

Notas

  • No se altera la lógica original. Solo se agregan docstrings/comentarios.
  • Pensado para Python 3.10+.

 138def merge_into_master(partial):

 159def _dedup_join_pipes(*texts: str, limit: int = 1200) -> str:
 160    """Une textos con '|', separa en frases por '|', hace trim + dedup y vuelve a unir."""
 161    seen, out = set(), []
 162    for t in texts:
 163        if not t:
 164            continue
 165        for part in t.split("|"):
 166            s = (part or "").strip()
 167            if s and s not in seen:
 168                seen.add(s)
 169                out.append(s)
 170    res = " | ".join(out)
 171    return res[:limit]

 246def extract_text_from_pdf(pdf_path: Path, enable_ocr: bool = S.ENABLE_OCR) -> str:
 247    doc = fitz.open(pdf_path)
 248    parts: List[str] = []
 249    for p in doc:
 250        if _page_has_text(p):
 251            parts.append(p.get_text("text"))
 252        else:
 253            if not enable_ocr: continue
 254            pix = p.get_pixmap(dpi=250)
 255            img = Image.frombytes("RGB", [pix.width, pix.height], pix.samples)
 256            parts.append(ocr_image(img))
 257    return "\n".join(parts).strip()

 259def extract_text_head(pdf_path: Path, enable_ocr: bool = S.ENABLE_OCR, max_chars: int = S.CTX_HEAD_PER_DOC) -> str:

 314class KBConfig:
 315    def __init__(self, path: Path):
 316        try:
 317            with open(path, "r", encoding="utf-8") as f:
 318                data = yaml.safe_load(f) or {}
 319        except FileNotFoundError:
 320            data = {}
 321        self.raw = data
 322        self.facets = data.get("facets", {})
 323        self.bundles = data.get("bundles", [])
 324        self._compiled = {}
 325        for facet, items in self.facets.items():
 326            self._compiled[facet] = []
 327            for it in items:
 328                pats = [re.compile(p, re.I) for p in it.get("patterns", [])]
 329                self._compiled[facet].append((it.get("label"), pats))
 330
 331    def match_facets(self, text: str) -> dict:
 332        text = _norm(text or "")
 333        res = {"familia": None, "procedimiento": None, "doc_role": None}
 334        for facet in ["familia", "procedimiento", "doc_role"]:
 335            for label, pats in self._compiled.get(facet, []):
 336                if any(p.search(text) for p in pats):
 337                    res[facet] = label
 338                    break
 339        return res
 340
 341    def find_bundle(self, familia: str, procedimiento: str) -> dict | None:
 342        for b in self.bundles:
 343            w = b.get("when", {})
 344            if w.get("familia") == familia and w.get("procedimiento") == procedimiento:
 345                return b
 346        return None

 365def index_knowledge_tree(kb_root: Path, collection_name: str) -> int:
 366    vs = _ensure_chroma(collection_name)
 367    added = 0
 368    existing_sources = set()
 369    try:
 370        got = vs._collection.get(include=["metadatas"])
 371        for md in (got.get("metadatas") or []):
 372            if md and md.get("source"): existing_sources.add(md["source"])
 373    except Exception:
 374        pass
 375    for pdf in kb_root.rglob("*.pdf"):
 376        rel = pdf.relative_to(kb_root).as_posix()
 377        if rel in existing_sources: continue
 378        meta_extra = derive_meta_from_kb_path_and_name(pdf)
 379        text = extract_text_from_pdf(pdf)
 380        if not text: continue
 381        meta = { "path": str(pdf), "source": rel, "collection": collection_name, **meta_extra }
 382        docs = to_documents(text, meta=meta)
 383        if docs:
 384            vs.add_documents(docs); added += len(docs)
 385    try: vs.persist()
 386    except Exception: pass
 387    return added

 389def _enforce_consistency_between_text_and_json(data_json: dict) -> dict:
 390    """Si la narrativa menciona riesgos, asegúrate de copiarlos al JSON si faltan."""
 391    try:
 392        r1 = (data_json.get("respuesta1") or {}).get("por_archivo") or []
 393        r2 = (data_json.get("respuesta2") or {}).get("por_archivo") or []
 394        by_file = { (it or {}).get("archivo"): it for it in r2 }
 395        for item in r1:
 396            stem = (item or {}).get("archivo")
 397            texto = ((item or {}).get("texto") or "").lower()
 398            if not stem or stem not in by_file: 
 399                continue
 400            j = (by_file[stem] or {}).get("json") or {}
 401            # Si ya hay riesgos, no tocar
 402            jr = j.get("riesgos") or []
 403            # Detecta “Riesgos del oferente:\nBAJO|MEDIO|ALTO: ...”
 404            bloques = re.findall(r"riesgos del oferente:\s*(.*)", texto, flags=re.I|re.S)
 405            if bloques and not jr:
 406                lines = [l.strip() for l in re.split(r"[\n•\-]+", bloques[0]) if l.strip()]
 407                out = []
 408                for l in lines:
 409                    m = re.match(r"(ALTO|MEDIO|BAJO)\s*:\s*(.+)", l, flags=re.I)
 410                    if m:
 411                        out.append({
 412                            "nivel": m.group(1).upper(),
 413                            "clausula": None,
 414                            "pagina": None,
 415                            "descripcion": m.group(2).strip()
 416                        })
 417                if out:
 418                    j["riesgos"] = out
 419                    by_file[stem]["json"] = j
 420        return data_json
 421    except Exception:
 422        return data_json

 513def _norm_key(s: str | None) -> str:
 514    s = ''.join(c for c in unicodedata.normalize("NFD", (s or "")) if unicodedata.category(c) != "Mn")
 515    s = s.lower()
 516    s = re.sub(r'[^a-z0-9]+', ' ', s)     # todo lo no alfanum a espacio
 517    s = re.sub(r'\b y \b', ' ', s)        # quitar " y "
 518    s = re.sub(r'\s+', ' ', s).strip()
 519    return s

 521def _canonize_fp(familia: str | None, procedimiento: str | None) -> tuple[str | None, str | None]:
 522    nf, np = _norm_key(familia), _norm_key(procedimiento)
 523    # normalizaciones suaves
 524    if np.startswith("menor"): np = "menor cuantia"
 525    if np.startswith("subasta inversa"): np = "subasta inversa electronica"
 526
 527    for fp in FAMILIA_PROCEDIMIENTO_VALIDOS:
 528        af, ap = _norm_key(fp["familia"]), _norm_key(fp["procedimiento"])
 529        if nf == af and np == ap:
 530            return fp["familia"], fp["procedimiento"]
 531    # Relajar si vino proc medio raro pero familia correcta
 532    for fp in FAMILIA_PROCEDIMIENTO_VALIDOS:
 533        if _norm_key(fp["familia"]) == nf and (not np or np in _norm_key(fp["procedimiento"])):
 534            return fp["familia"], fp["procedimiento"]
 535    return None, None

 559def _extract_vendor_names(oferta_text: str) -> list[str]:
 560    if not oferta_text: 
 561        return []
 562    cands = []
 563
 564    # 1) Líneas con etiqueta (Razón social:, Proveedor:, etc.)
 565    for line in oferta_text.splitlines():
 566        m = RS_NEAR.search(line)
 567        if m:
 568            seg = m.group(1).strip()
 569            # Si recortaron con “Entre … y …”, toma el primer bloque “hasta , o ;”
 570            seg = re.split(r"[;,]| y ", seg, maxsplit=1)[0].strip()
 571            cands.append(seg)
 572
 573    # 2) Candidatos por forma (mayúsculas + terminaciones típicas)
 574    for m in RS_CAND_RE.finditer(oferta_text):
 575        cands.append(m.group(1).strip())
 576
 577    # 3) Limpieza básica y dedup
 578    out, seen = [], set()
 579    for c in cands:
 580        c = re.sub(r"\s{2,}", " ", c).strip(" :–—-")
 581        if len(c) < 4: 
 582            continue
 583        k = c.lower()
 584        if k not in seen:
 585            seen.add(k); out.append(c)
 586    return out[:5]

 733def _find_numeric_near_keywords(text: str, value_re: re.Pattern, keywords: tuple[str, ...], window: int = 120) -> Optional[int]:
 734    if not text: return None
 735    low = text.lower()
 736    for m in value_re.finditer(text):
 737        start, end = m.span()
 738        left = max(0, start - window)
 739        right = min(len(text), end + window)
 740        ctx = low[left:right]
 741        if any(k in ctx for k in keywords):
 742            val = _to_int_safe(m.group(1) if m.lastindex else m.group(0))
 743            if val is not None: return val
 744    return None

1221def _clean_filter(f):

1223        out = {}
1224        for k, v in f.items():

1226                cleaned = [_clean_filter(x) for x in v if _clean_filter(x) is not None]
1227                if cleaned: out[k] = cleaned
1228            elif isinstance(v, dict) and "$eq" in v and v["$eq"] is None:
1229                continue
1230            else:

1232                if cv is not None and cv != {} and cv != []:
1233                    out[k] = cv
1234        return out or None
